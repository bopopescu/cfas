## Policies
# List
curl -X GET http://localhost:8000/v3/policies/ | python -mjson.tool
# Retrieve
curl -X GET http://localhost:8000/v3/policies/9/ | python -mjson.tool
# Create
curl -X POST -H "Content-Type: application/json" -d '{ "description": "test" }' http://localhost:8000/v3/policies/ | python -mjson.tool
--
{
    "description": "test",
    "id": 13
}
# Delete
curl -X DELETE http://localhost:8000/v3/policies/13/ | python -mjson.tool


## And Rules
# List
curl -X GET http://localhost:8000/v3/policies/and_rules/ | python -mjson.tool
# List with filter by Policy
curl -X GET http://localhost:8000/v3/policies/and_rules/?policy=10 | python -mjson.tool
# Retrieve
curl -X GET http://localhost:8000/v3/policies/and_rules/1056/ | python -mjson.tool
# Create
curl -X POST -H "Content-Type: application/json" -d '{ "policy": 9, "description": "identity:test", "enabled": false, "conditions": [] }' http://localhost:8000/v3/policies/and_rules/ | python -mjson.tool
--
{
    "conditions": [],
    "description": "identity:test",
    "enabled": false,
    "id": 1432,
    "policy": 9
}
# Patch
curl -X PATCH -H "Content-Type: application/json" -d '{ "policy": 9, "description": "identity:test", "enabled": false, "conditions": [1109] }' http://localhost:8000/v3/policies/and_rules/1432/ | python -mjson.tool
# Delete
curl -X DELETE http://localhost:8000/v3/policies/and_rules/1432/ | python -mjson.tool


## Conditions

--

    "attribute": "xxx",
