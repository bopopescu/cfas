## API Calls Examples

## Policies
# List
curl -v -X GET http://localhost:8000/v3/policies/ | python -mjson.tool
# Retrieve
curl -v -X GET http://localhost:8000/v3/policies/1/ | python -mjson.tool
# Create
curl -v -X POST -H "Content-Type: application/json" -d @pol.json http://localhost:8000/v3/policies/ | python -mjson.tool
# Update
curl -v -X PATCH -H "Content-Type: application/json" -d @pol.json http://localhost:8000/v3/policies/1/ | python -mjson.tool
# Delete
curl -X DELETE http://localhost:8000/v3/policies/13/ | python -mjson.tool


## And Rules
# List
curl -X GET http://localhost:8000/v3/policies/and_rules/ | python -mjson.tool
# List with filter by Policy
curl -X GET http://localhost:8000/v3/policies/and_rules/?policy=10 | python -mjson.tool
# Retrieve
curl -X GET http://localhost:8000/v3/policies/and_rules/1056/ | python -mjson.tool
# Create
curl -X POST -H "Content-Type: application/json" -d '{ "policy": 1, "description": "identity:test", "enabled": false, "conditions": [] }' http://localhost:8000/v3/policies/and_rules/ | python -mjson.tool
# Patch
curl -X PATCH -H "Content-Type: application/json" -d '{ "policy": 1, "description": "identity:test", "enabled": false, "conditions": [1109] }' http://localhost:8000/v3/policies/and_rules/1/ | python -mjson.tool
# Delete
curl -X DELETE http://localhost:8000/v3/policies/and_rules/1/ | python -mjson.tool


## Conditions
# List
curl -X GET http://localhost:8000/v3/policies/conditions/ | python -mjson.tool
# Retrieve
curl -X GET http://localhost:8000/v3/policies/conditions/1/ | python -mjson.tool
# Create
curl -X POST -H "Content-Type: application/json" -d '{ "attribute": "xxx", "operator": "=", "value": "111", "description": "xxx=111" }' http://localhost:8000/v3/policies/conditions/ | python -mjson.tool
# Delete
curl -X DELETE http://localhost:8000/v3/policies/conditions/1107/ | python -mjson.tool


## Search
curl -v -X POST -H "Content-Type: application/json" -d @search.json http://localhost:8000/v3/policies/1/search | python -mjson.tool
